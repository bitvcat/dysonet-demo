21local function _newAgent(fd, addr)
22 local secretKey = crypt.randomkey()
23 local agent = {
24 fd = fd,
25 addr = addr,
26 randomCode = crypt.hexencode(crypt.randomkey()),
27 secretKey = secretKey,
28 publicKey = crypt.dhexchange(secretKey),
29 encryptKey = "",
30 handshake = 0
31 }
32 return agent
33end

57function handler.onAccept(fd, addr)

72function handler.onConnect(fd, addr)
73 socket_start(fd)
74
75 assert(not connection[fd])
76 clientNumber = clientNumber + 1
77 local agent = _newAgent(fd, addr)
78 connection[fd] = agent
79 skynet.error(string.format("%s connected as %d", addr, fd))
80
81 -- send to watchdog
82 skynet.send(watchdog, "lua", "Client", "onConnect", fd, addr, gateNode, gateAddr, protocol)

84 skynet.timeout(0, function()

86 local ok, msg = socket_read(fd)

97function handler.onMessage(fd, msg)
98 local agent = connection[fd]
99 if not agent then
100 skynet.error(string.format("Invalid agent, fd = %d", fd or -1))
101 return
102 end
103
104 if agent.handshake == 0 then
105 local _, hellostr = protobuf.decode_message(msg)
106 assert(hellostr == "client hello", hellostr)
107
108 -- D-H exchange
109 local hexstr = agent.randomCode .. "|" .. crypt.hexencode(agent.publicKey)
110 local msg = protobuf.encode_message(nil, hexstr)
111 socket_write(fd, msg)
112 agent.handshake = 1
113 elseif agent.handshake == 1 then
114 local _, hexstr = protobuf.decode_message(msg)
115 local randomHex, cPublicHex = string.match(hexstr, "(.+)|(.+)")
116 local randomDes, cPublicKey = crypt.hexdecode(randomHex), crypt.hexdecode(cPublicHex)
117 local encryptKey = crypt.dhsecret(cPublicKey, agent.secretKey)
118 if agent.randomCode == crypt.desdecode(encryptKey, randomDes) then
119 agent.encryptKey = encryptKey
120 agent.handshake = 2
121
122 local msg = protobuf.encode_message(nil, "server done")
123 socket_write(fd, msg)
124 else
125 handler.onClose(fd, "validationFail")
126 return
127 end
128 elseif agent.handshake == 2 then
129 local opname, args, session = protobuf.decode_message(msg)
130 if #opname > 0 then
131 skynet.send(watchdog, "lua", "Client", "onMessage", fd, opname, args, session)
132 end
133 else
134 handler.onClose(fd, "unauthLink")
135 end
136end

140 skynet.send(watchdog, "lua", "Client", "onClose", fd, reason)
141end

144local LUA = {}
145function LUA.open(conf)
146 gateIp = conf.ip or "::"
147 gatePort = conf.port
148 watchdog = conf.watchdog
149 gateNode = skynet.getenv("name")
150 gateAddr = skynet.self()
151 protocol = conf.protocol
152 protobuf.start({ pbfile = "assets/proto/all.pb" })
153
154 if not conf.isSlave then
155 table.insert(gates, skynet.self())
156
157 -- slave gates
158 conf.isSlave = true
159 local slaveNum = conf.slaveNum or 0
160 for i = 1, slaveNum, 1 do
161 local slaveGate = skynet.newservice("gate_tcp")
162 skynet.call(slaveGate, "lua", "open", conf)
163 table.insert(gates, slaveGate)
164 end
165
166 -- listen
167 local id = assert(socket.listen(gateIp, gatePort))
168 skynet.error(string.format("Listen tcp gate at %s:%s", gateIp, gatePort))
169 socket.start(id, function(fd, addr)
170 skynet.error(string.format("%s accept as %d", addr, fd))
171 handler.onAccept(fd, addr)
172 end)
173 end
174 skynet.retpack()
175end
