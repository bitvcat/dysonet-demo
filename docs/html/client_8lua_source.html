39function Client:getGateConfig(protocol)
40 local skynet = dysonet.skynet
41 local gateConf = {
42 port = tonumber(skynet.getenv(protocol.."_port")),
43 slaveNum = tonumber(skynet.getenv("gate_slave_num")),
44 watchdog = skynet.self(),
45 protocol = protocol
46 }
47 local suffix = protocol == "wss" and "ws" or protocol
48 return "gate_"..suffix, gateConf
49end

51function Client:_open(protocol)
52 local skynet = dysonet.skynet
53 local gateName, conf = self:getGateConfig(protocol)
54 assert(gateName)
55
56 local gate = assert(skynet.newservice(gateName))
57 skynet.call(gate, "lua", "open", conf)
58 return gate
59end
