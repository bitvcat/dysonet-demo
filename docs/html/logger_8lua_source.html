35--- @brief 日志模块初始化
36--- @details 注册<tt>text</tt>类型的消息处理，并启动 num 个 xlogger C服务。
37--- @param string levelname 限制日志等级
38--- @param integer num xlogger 服务数量
39function logger.init(levelname, num)
40 logger.wraps = {
41 DEBUG = "[DEBUG]",
42 INFO = "[INFO]",
43 WARN = "[WARN]",
44 ERROR = "[ERROR]",
45 FATAL = "[FATAL]",
46 }
47 logger.level = logger[levelname] or logger.INFO
48
49 skynet.register_protocol {
50 name = "text",
51 id = skynet.PTYPE_TEXT,
52 pack = function(text) return text end,
53 unpack = function(buf, sz) return skynet.tostring(buf, sz) end,
54 }
55 if not logger.addresses then
56 logger.msgtable = { "", "", ""}
57 logger.addresses = {}
58 num = num or 4
59 local logpath = skynet.getenv("logpath") or "log"
60 for i = 1, num, 1 do
61 local addr = assert(skynet.launch("xlogger", logpath))
62 logger.addresses[i] = addr
63 end
64 end
65end

67--- @brief 根据日志文件名获取对应的服务地址
68--- @details 使用 elfhash 算法获取 filename 的哈希值。
69--- @param string filename 日志文件名
70--- @return 日志服务地址
71function logger.getAddr(filename)
72 assert(#filename > 0)
73 assert(#logger.addresses > 0)
74 local hashkey = lutil.elfhash(filename)
75 local index = (hashkey % #logger.addresses) + 1
76 return logger.addresses[index]
77end

79function logger.format(fmt, ...)
80 local msg

82 msg = fmt

87end

89--- @brief 格式化日志消息并写入到文件中
90--- @param string levelname 指定日志等级
91--- @param string filename 日志文件名
92--- @param string fmt 日志消息格式化串
93function logger.logf(levelname, filename, fmt, ...)
94 if logger[levelname] < logger.level then
95 return
96 end
97
98 local msgtable = logger.msgtable
99 local addr = logger.getAddr(filename)
100 msgtable[1] = filename
101 msgtable[2] = logger.wraps[levelname] or levelname
102 msgtable[3] = logger.format(fmt, ...)
103 skynet.send(addr, "text", table.concat(msgtable, " "))
104end

106--- @brief 打印运行时日志，支持打印 table
107--- @param vars ... 要打印的值列表
108function logger.print(...)
109 local t = { ... }
110 for i, value in ipairs(t) do
111 t[i] = type(value) == "table" and table.dump(value) or value
112 end
113 local info = debug.getinfo(2)
114 local prefix = (info.source or "?") .. ":" .. info.currentline
115 skynet.error("[logger.print]", prefix, table.concat(t,"\n"))
116end
