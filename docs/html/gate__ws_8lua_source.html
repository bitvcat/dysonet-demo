79local LUA = {}
80function LUA.open(conf)
81 gateIp = conf.ip or "::"
82 gatePort = conf.port
83 watchdog = conf.watchdog
84 protocol = conf.protocol
85 gateNode = skynet.getenv("name")
86 gateAddr = skynet.self()
87
88 if not conf.isSlave then
89 table.insert(gates, skynet.self())
90
91 -- slave gates
92 conf.isSlave = true
93 local slaveNum = conf.slaveNum or 0
94 for i = 1, slaveNum, 1 do
95 local slaveGate = skynet.newservice("gate_ws")
96 skynet.call(slaveGate, "lua", "open", conf)
97 table.insert(gates, slaveGate)
98 end
99
100 -- listen
101 local id = assert(socket.listen(gateIp, gatePort))
102 skynet.error(string.format("Listen websocket gate at %s:%s protocol:%s", gateIp, gatePort, protocol))
103 socket.start(id, function(fd, addr)
104 skynet.error(string.format("%s accept as %d", addr, fd))
105 accept(fd, addr)
106 end)
107 end
108 skynet.retpack()
109end
